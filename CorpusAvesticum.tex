   function modifySorting()
      local f = io.open("passages.idx", "r+")
      local content = f:read("*all")
      f:seek('set')
      -- f:close()
      content = string.gsub(content, "\\gls {([^}]*)}~?(%d+)%.(%d+)",
      function(a,b,c) return
         string.format("%s%03d.%03d@%s~%d.%d", a, b, c, a, b, c) end)
      f:write(content)
      f:close()
   end
